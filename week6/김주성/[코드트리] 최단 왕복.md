### 문제
https://www.codetree.ai/missions/8/problems/shortest-round-trip?&utm_source=clipboard&utm_medium=text

### 코드
~~~
n, m = map(int, input().split())

distances = [[-1 for _ in range(n+1)] for _ in range(n+1)]

for i in range(1, n+1):
    distances[i][i] = 0

graph = [[] for _ in range(n+1)]

for _ in range(m):
    s, e, d = map(int, input().split())
    if distances[s][e] == -1 or distances[s][e] > d:
        distances[s][e] = d

for k in range(1, n+1):
    for i in range(1, n+1):
        for j in range(1, n+1):
            if distances[i][k] == -1 or distances[k][j] == -1:
                continue
            if distances[i][j] == -1 or distances[i][j] > distances[i][k]+distances[k][j]:
                distances[i][j] = distances[i][k]+distances[k][j]

answer = 1e9

for i in range(1, n+1):
    for j in range(1, n+1):
        if i == j:
            continue
        if distances[i][j] != -1 and distances[j][i] != -1:
            answer = min(answer, distances[i][j]+distances[j][i])
print(answer)
~~~
